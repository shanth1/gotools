# Examples

```go
type Config struct {
	ConfigPath string `flag:"config" usage:"Path to the YAML config file"`

	LogLevel string `flag:"log-level" default:"info" usage:"Logging level (debug, info, error)"`
	Port     int    `flag:"port"      default:"8080" usage:"TCP port to listen on"`
	DevMode  bool   `flag:"dev-mode"  default:"false" usage:"Enable development mode with more verbose logging"`
}

func main() {
	cfg := &Config{}
	if err := flags.RegisterFromStruct(cfg); err != nil {
		log.Fatalf("register flag: %v", err)
	}
}
```
