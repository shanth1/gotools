# Examples

```go
func main() {
	logger := log.New(log.WithService("test"), log.WithLevel(log.LevelInfo))

	logger.Trace().Msg("This message will not be logged")
	logger.Debug().Msg("This message will not be logged")
	logger.Info().Msgf("%s level msg", "Info")
	logger.Warn().Bool("bool", true).Msgf("Warn level msg")
	logger.Error().Err(errors.New("some error")).Msgf("Error level msg")
	logger.Fatal().Err(errors.New("error code 1")).Msgf("Fatal")
}
```
